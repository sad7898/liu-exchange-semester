21 function hasuniquerep(@nospecialize t)
22 # typeof(Bottom) is special since even though it is a leaftype,
23 # at runtime, it might be Type{Union{}} instead, so don't attempt inference of it
24 t === typeof(Union{}) && return false
25 t === Union{} && return true

27 iskindtype(typeof(t)) || return true # non-types are always compared by egal in the type system
28 isconcretetype(t) && return true # these are also interned and pointer comparable
