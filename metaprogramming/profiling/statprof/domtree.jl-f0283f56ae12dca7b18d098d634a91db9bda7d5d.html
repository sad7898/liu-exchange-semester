22 """
23 Checks if bb1 dominates bb2.
24 bb1 and bb2 are indexes into the CFG blocks.
25 bb1 dominates bb2 if the only way to enter bb2 is via bb1.
26 (Other blocks may be in between, e.g bb1->bbX->bb2).
27 """
28 function dominates(domtree::DomTree, bb1::Int, bb2::Int)
29 bb1 == bb2 && return true
30 target_level = domtree.nodes[bb1].level
31 source_level = domtree.nodes[bb2].level
32 source_level < target_level && return false
33 for _ in (source_level - 1):-1:target_level
34 bb2 = domtree.idoms[bb2]
35 end
36 return bb1 == bb2
37 end

74 function naive_idoms(cfg::CFG)
75 nblocks = length(cfg.blocks)
76 # The extra +1 helps us detect unreachable blocks below
77 dom_all = BitSet(1:nblocks+1)
78 dominators = BitSet[n == 1 ? BitSet(1) : copy(dom_all) for n = 1:nblocks]
79 changed = true
80 while changed
81 changed = false
82 for n = 2:nblocks
83 if isempty(cfg.blocks[n].preds)
84 continue
85 end
86 firstp, rest = Iterators.peel(Iterators.filter(p->p != 0, cfg.blocks[n].preds))
87 new_doms = copy(dominators[firstp])
88 for p in rest
89 intersect!(new_doms, dominators[p])
90 end
91 push!(new_doms, n)
92 changed = changed || (new_doms != dominators[n])
93 dominators[n] = new_doms
94 end
95 end
96 # Compute idoms
97 idoms = fill(0, nblocks)
98 for i = 2:nblocks
99 if dominators[i] == dom_all
100 idoms[i] = 0
101 continue
102 end
103 doms = collect(dominators[i])
104 for dom in doms
105 i == dom && continue
106 hasany = false
107 for p in doms
108 if p !== i && p !== dom && dom in dominators[p]
109 hasany = true; break
110 end
111 end
112 hasany && continue
113 idoms[i] = dom
114 end
115 end
116 idoms
117 end

213 """
214 Matches the snca_compress algorithm in Figure 2.8 of [LG05], with the
215 modification suggested in the paper to use `last_linked` to determine
216 whether an ancestor has been processed rather than storing `0` in the
217 ancestor array.
218 """
219 function snca_compress!(state::Vector{Node}, ancestors::Vector{DFSNumber},
220 v::DFSNumber, last_linked::DFSNumber)
221 u = ancestors[v]
222 @assert u < v
223 if u >= last_linked
224 snca_compress!(state, ancestors, u, last_linked)
225 if state[u].label < state[v].label
226 state[v] = Node(state[v].semi, state[u].label)
227 end
228 ancestors[v] = ancestors[u]
229 end
230 nothing
231 end

233 function snca_compress_worklist!(
234 state::Vector{Node}, ancestors::Vector{DFSNumber},
235 v::DFSNumber, last_linked::DFSNumber)
236 # TODO: There is a smarter way to do this
237 u = ancestors[v]
238 worklist = Tuple{DFSNumber, DFSNumber}[(u,v)]
239 @assert u < v
240 while !isempty(worklist)
241 u, v = last(worklist)
242 if u >= last_linked
243 if ancestors[u] >= last_linked
244 push!(worklist, (ancestors[u], u))
245 continue
246 end
247 if state[u].label < state[v].label
248 state[v] = Node(state[v].semi, state[u].label)
249 end
250 ancestors[v] = ancestors[u]
251 end
252 pop!(worklist)
253 end

D = DFS(cfg, BBNumber(1))
270 # `label` is initialized to the identity mapping (though
271 # the paper doesn't make that clear). The rational for this is Lemma
272 # 2.4 in [LG05] (i.e. Theorem 4 in ). Note however, that we don't
273 # ever look at `semi` until it is fully initialized, so we could leave
274 # it uninitialized here if we wanted to.
275 state = Node[ Node(typemax(DFSNumber), w) for w in preorder(D) ]
276 # Initialize idoms to parents. Note that while idoms are eventually
277 # BB indexed, we keep it DFS indexed until a final post-processing
278 # pass to avoid extra memory references during the O(N^2) phase below.
279 idoms_dfs = copy(D.parents)
280 # We abuse the parents array as the ancestors array.
281 # Semi-NCA does not look at the parents array at all.
282 # SLT would, but never simultaneously, so we could still
283 # do this.
284 ancestors = D.parents
285 for w::DFSNumber ∈ reverse(_drop(preorder(D), 1))
286 # LLVM initializes this to the parent, the paper initializes this to
287 # `w`, but it doesn't really matter (the parent is a predecessor,
288 # so at worst we'll discover it below). Save a memory reference here.
289 semi_w = typemax(DFSNumber)
290 for v ∈ cfg.blocks[D.numbering[w]].preds
291 # For the purpose of the domtree, ignore virtual predecessors
292 # into catch blocks.
293 v == 0 && continue
294 vdfs = D.reverse[v]
295 # Ignore unreachable predecessors
296 vdfs == 0 && continue
297 last_linked = DFSNumber(w + 1)
298 # N.B.: This conditional is missing from the psuedocode
299 # in figure 2.8 of [LG05]. It corresponds to the
300 # `ancestor[v] != 0` check in the `eval` implementation in
301 # figure 2.6
302 if vdfs >= last_linked
303 # For performance, if the number of ancestors is small
304 # avoid the extra allocation of the worklist.
305 if length(ancestors) <= 32
306 snca_compress!(state, ancestors, vdfs, last_linked)
307 else
308 snca_compress_worklist!(state, ancestors, vdfs, last_linked)
309 end
310 end
311 semi_w = min(semi_w, state[vdfs].label)
312 end
313 state[w] = Node(semi_w, semi_w)
314 end
315 for v ∈ _drop(preorder(D), 1)
316 idom = idoms_dfs[v]
317 vsemi = state[v].semi
318 while idom > vsemi
319 idom = idoms_dfs[idom]
320 end
321 idoms_dfs[v] = idom
322 end
323 # Reexpress the idom relationship in BB indexing
324 idoms_bb = Int[ (i == 1 || D.reverse[i] == 0) ? 0 : D.numbering[idoms_dfs[D.reverse[i]]] for i = 1:length(cfg.blocks) ]
