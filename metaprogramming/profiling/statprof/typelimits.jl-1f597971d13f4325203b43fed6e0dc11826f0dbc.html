38 function is_derived_type(@nospecialize(t), @nospecialize(c), mindepth::Int)
39 if t === c
40 return mindepth <= 1
41 end
42 if isa(c, Union)
43 # see if it is one of the elements of the union
44 return is_derived_type(t, c.a, mindepth) || is_derived_type(t, c.b, mindepth)
45 elseif isa(c, UnionAll)

47 # also handle the var here, since this construct bounds the mindepth to the smallest possible value
48 return is_derived_type(t, c.var.ub, mindepth) || is_derived_type(t, c.body, mindepth)
49 elseif isa(c, DataType)
50 if mindepth > 0
51 mindepth -= 1

53 if isa(t, DataType)

55 super = supertype(c)

57 t === super && return true
58 super = supertype(super)
59 end
60 end
